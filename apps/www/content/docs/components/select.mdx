## Usage

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select"
```

```tsx
<Select>
  <SelectTrigger className="w-[180px]">
    <SelectValue placeholder="Theme" />
  </SelectPopover>
    <SelectContent>
      <SelectItem id="light">Light</SelectItem>
      <SelectItem id="dark">Dark</SelectItem>
      <SelectItem id="system">System</SelectItem>
    </SelectContent>
  </SelectPopover>
</Select>
```
