## Usage

```tsx
import { Item, ListBox } from "@/components/ui/list-box"
```

```tsx
<ListBox aria-label="Example list box">
  <Item>Option 1</Item>
  <Item>Option 2</Item>
  <Item>Option 3</Item>
</ListBox>
```
